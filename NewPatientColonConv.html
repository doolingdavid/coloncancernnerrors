    """Need to loop through threshold values high to low, accumulate
    the true positive rates and false positive rates into a list
    and then turn it all into three arrays and return them."""
    
    fprlist = []
    tprlist = []

    duh = list(np.arange(1.1, -.01, -.01))

    for i in range(len(duh)):
        print(duh[i], type(float(duh[i])))
        prob0pos = (dfroc['pp0'] >= float(duh[i]))  # this value here is the threshold
        prob0neg = (dfroc['pp0'] < float(duh[i]))
        surv0pos = (dfroc['true0'] == 1)
        surv0neg = (dfroc['true0'] != 1)

        fp = dfroc[prob0pos & surv0neg].shape[0]

        tpr = tp/float(tp + fn)

        fpr = fp/float(fp + tn)
        
        #print "False positive rate: ", fpr
        
        fprlist.append(fpr)
        
        print("Confusion matrix:")
        print(tp, fn)
        print(fp, tn)
        
    
        
    return tprlist, fprlist
